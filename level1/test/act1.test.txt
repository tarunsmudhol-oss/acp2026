@test "Normal temperatures" {
  run ./act1.exe <<EOF
25
26
27
28
29
30
31
EOF
  [[ "$output" == *"Highest Temperature = 31.00"* ]]
  [[ "$output" == *"Lowest Temperature = 25.00"* ]]
}
