#!/usr/bin/env bats

@test "test 1" {
   run bash -c './demo1.exe <<EOF

5
10.5
25.2
7.8
30.1
18.6
EOF'

echo $output >&3

[[ "$output" = *"The Maxium Index is = 30.10"* ]]
}

@test "test 2" {
   run bash -c './demo1.exe <<EOF

4
-10.5
-2.3
-50.7
-1.1
 EOF'

echo $output >&3

[[ "$output" = *"The Maxium Index is = -1.10"* ]]
}

@test "test 3" {
   run bash -c './demo1.exe <<EOF

9.9
9.9

EOF'
