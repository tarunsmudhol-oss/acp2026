#!/usr/bin/env bats

@test "test 1 - swap arrays with 3 elements" {
run bash -c './demo5.exe <<EOF
3
1 2 3
4 5 6
EOF'

echo "$output" >&3

[[ "$output" = *"Before swapping:"* ]]
[[ "$output" = *"Array 1: 1 2 3"* ]]
[[ "$output" = *"Array 2: 4 5 6"* ]]
[[ "$output" = *"After swapping:"* ]]
[[ "$output" = *"Array 1: 4 5 6"* ]]
[[ "$output" = *"Array 2: 1 2 3"* ]]
}
