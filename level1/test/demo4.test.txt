#!/usr/bin/env bats

@test "test 1 - destination found (single match)" {
run bash -c './demo4.exe <<EOF
101
Delhi
50
102
Mumbai
40
103
Chennai
30
104
Kolkata
20
Mumbai
EOF'

echo "$output" >&3

[[ "$output" = *"Flight Number: 102"* ]]
}

@test "test 2 - destination not found" {
run bash -c './demo4.exe <<EOF
101
Delhi
50
102
Mumbai
40
103
Chennai
30
104
Kolkata
20
Hyderabad
EOF'

echo "$output" >&3

[[ "$output" = *"No flights available to Hyderabad"* ]]
}

@test "test 3 - multiple flights same destination" {
run bash -c './demo4.exe <<EOF
101
Delhi
50
102
Mumbai
40
103
Mumbai
25
104
Kolkata
20
Mumbai
EOF'

echo "$output" >&3

[[ "$output" = *"Flight Number: 102"* ]]
[[ "$output" = *"Flight Number: 103"* ]]
}

@test "test 4 - case sensitivity check" {
run bash -c './demo4.exe <<EOF
101
Delhi
50
102
Mumbai
40
103
Chennai
30
104
Kolkata
20
mumbai
EOF'

echo "$output" >&3

[[ "$output" = *"No flights available to mumbai"* ]]
}

@test "test 5 - first flight match" {

@test "test 7 - zero available seats" {
run bash -c './demo4.exe <<EOF
