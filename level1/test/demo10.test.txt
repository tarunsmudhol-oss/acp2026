@test "Test with 1 student" {
  run bash -c "./demo10.exe <<EOF
1
101
Tarun
89
EOF
"
  [[ "$output" == *"101"* ]]
  [[ "$output" == *"Tarun"* ]]
  [[ "$output" == *"89.00"* ]]
}

@test "Test with 2 students" {
  run bash -c "./demo10.exe <<EOF
2
101
Tarun
89
102
Prajwal
95
EOF
"
  [[ "$output" == *"101"* ]]
  [[ "$output" == *"102"* ]]
  [[ "$output" == *"Prajwal"* ]]
}
