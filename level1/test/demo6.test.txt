#!/usr/bin/env bats

@test "test 1 - normal input (5 elements)" {
run bash -c './demo6.exe <<EOF
5
1 2 3 4 5
EOF'

echo "$output" >&3

[[ "$output" = *"Enter 5 elements:"* ]]
[[ "$output" = *"1 2 3 4 5"* ]]
[[ "$output" = *"Memory successfully freed and pointer set to NULL."* ]]
}
