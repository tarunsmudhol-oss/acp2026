#!/usr/bin/env bats

@test "test 1 - rectangle 1 is largest" {
run bash -c './demo2.exe <<EOF
Rect1
10
5
Rect2
4
5
Rect3
3
6
EOF'

echo "$output" >&3
[[ "$output" == *"Largest Rectangle: Rect1"* ]]
}

@test "test 2 - rectangle 2 is largest" {
run bash -c './demo2.exe <<EOF

R2

2
R3
4
4
EOF'

[[ "$output" == *"Largest Rectangle: R2"* ]]
}
