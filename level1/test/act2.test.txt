@test " 1 Units less than 50 (free scheme)" {
  run ./act2.exe <<EOF
40
5
EOF
  [[ "$output" == *"Total Bill: Rup 0.00"* ]]
  [[ "$output" == *"Eligible for free unit consumption scheme"* ]]
}

@test "2 Units exactly 50" {
  run ./act2.exe <<EOF
50
2	
EOF
  [[ "$output" == *"Total Bill: Rup 100.00"* ]]
  [[ "$output" == *"Normal billing applied"* ]]
}

@test " 3 Units between 50 and 100" {
  run ./act2.exe <<EOF

3
EOF
  [[ "$output" == *"Total Bill: Rup 240.00"* ]]
  [[ "$output" == *"Normal billing applied"* ]]
}
